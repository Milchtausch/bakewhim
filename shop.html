  <script>

    let cart = [];


    function addToCart(name, price) {

      const existing = cart.find(item => item.name === name);

      if (existing) {

        existing.quantity += 1;

      } else {

        cart.push({ name, price, quantity: 1 });

      }

      updateCart();

      toggleCart(true);

      localStorage.setItem("cart", JSON.stringify(cart));

    }


    function updateCart() {

      const cartItems = document.getElementById("cart-items");

      cartItems.innerHTML = "";

      let total = 0;


      cart.forEach((item, index) => {

        total += item.price * item.quantity;

        const el = document.createElement("div");

        el.innerHTML = `

          <strong>${item.name}</strong><br>

          ${item.price.toFixed(2)} CHF x ${item.quantity}

          <br>

          <button onclick="changeQty(${index}, 1)">+</button>

          <button onclick="changeQty(${index}, -1)">−</button>

          <button onclick="removeItem(${index})">Entfernen</button>

          <hr>

        `;

        cartItems.appendChild(el);

      });


      document.getElementById("total").innerText = total.toFixed(2);

      const checkoutBtn = document.getElementById("checkout-btn");

      checkoutBtn.disabled = cart.length === 0;

    }


    function changeQty(index, change) {

      cart[index].quantity += change;

      if (cart[index].quantity <= 0) cart.splice(index, 1);

      updateCart();

    }


    function removeItem(index) {

      cart.splice(index, 1);

      updateCart();

    }


    function toggleCart(forceOpen = null) {

      const cartEl = document.getElementById("cart");

      if (forceOpen === true) cartEl.style.display = "block";

      else if (forceOpen === false) cartEl.style.display = "none";

      else cartEl.style.display = cartEl.style.display === "none" ? "block" : "none";

    }

  </script>
